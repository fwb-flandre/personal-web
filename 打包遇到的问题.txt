	①加入对 woff 和 ttf 的检索
      {
        test: /\.(woff2?|woff|eot|ttf|otf)(\?.*)?$/,
        loader: 'url-loader',
        options: {
          limit: 10000,
          name: utils.assetsPath('fonts/[name].[hash:7].[ext]')
        }
      }
	②修改 publicPath 改为 '../../'
    if (options.extract) {
      return ExtractTextPlugin.extract({
        use: loaders,
        fallback: 'vue-style-loader',
        // 解决 element-ui 字体丢失
        publicPath: '../../'
      })
    } else {
      return ['vue-style-loader'].concat(loaders)
    }
